def self.json_load_file(filespec, opts = {})
  if RUBY_VERSION < '3.0.0'
    JSON.parse(File.read(filespec), opts)
  else
    JSON.load_file(filespec, opts)
  end
end

def self.tmpfile(name, url)
  tmp = Tempfile.new(name)
  tmp.write(Net::HTTP.get(URI(url)))
  tmp.close
  tmp
end
